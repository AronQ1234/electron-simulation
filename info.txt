export function ElectronFlowSim({
  voltage = 5,
  temp = 25,
  humidity = 0,
  V0_eV = 1.7,
  d_nm = 2.0,
  E_eV = 0.5,
  J_ex = 0.08,
  M_FMI = 1.0,
  particleCount = 220,
}) {
  // numeric domain (nm)
  const left_nm = -2 * d_nm;
  const right_nm = 3 * d_nm;
  const barrier_start = 0;
  const barrier_end = d_nm;

  const KB_EV = 8.617333e-5;

  // effective energy (E + thermal + voltage contribution)
  const E_eff = useMemo(() => {
    const vBoost = 0.02 * (Number(voltage || 0) - 1); // tuneable
    const tBoost = KB_EV * (Number(temp || 0));
    return Number(E_eV || 0) + vBoost + tBoost;
  }, [E_eV, voltage, temp]);

  const humidityFactor = useMemo(() => {
    const h = Math.max(0, Math.min(100, Number(humidity || 0)));
    return 1 - (h / 100) * 0.3;
  }, [humidity]);

  // Transmission probabilities (apply humidity factor)
  const Tvals = useMemo(() => {
    let tUp = 1e-12, tDown = 1e-12;
    try {
      if (typeof PH.calculateTunnelingProbabilitiesWKB === "function") {
        const res = PH.calculateTunnelingProbabilitiesWKB(V0_eV, E_eff, d_nm, J_ex, M_FMI);
        tUp = Number(res.T_up ?? res.Tup ?? 0);
        tDown = Number(res.T_down ?? res.Tdown ?? 0);
      } else {
        const DeltaE = typeof PH.spinSplitting === "function" ? PH.spinSplitting(J_ex, M_FMI) : J_ex * M_FMI;
        const V_up = V0_eV - DeltaE / 2.0;
        const V_down = V0_eV + DeltaE / 2.0;
        const d_m = d_nm * 1e-9;
        tUp = PH.wkbTunneling ? PH.wkbTunneling(V_up, d_m, E_eff) : 1e-12;
        tDown = PH.wkbTunneling ? PH.wkbTunneling(V_down, d_m, E_eff) : 1e-12;
      }
    } catch (err) {
      console.warn("Tvals calc failed:", err);
    }
    tUp = Math.max(0, Math.min(1, tUp * humidityFactor));
    tDown = Math.max(0, Math.min(1, tDown * humidityFactor));
    return { T_up: tUp, T_down: tDown };
  }, [V0_eV, d_nm, E_eff, J_ex, M_FMI, humidityFactor]);

  // particles stored in ref for high-rate updates (no rerenders per frame)
  const particlesRef = useRef(
    new Array(particleCount).fill(0).map((_, i) => {
      const spin = Math.random() < 0.5 ? "up" : "down";
      const z = (Math.random() - 0.5) * 1.2;
      const x = left_nm + Math.random() * (Math.abs(left_nm) * 0.25);
      return { id: i, x, z, spin, decided: false, transmitted: false, speed: 0.4 + Math.random() * 0.8 };
    })
  );

  // counts shown in UI (update from frame loop)
  const [counts, setCounts] = useState({ transmitted: 0, reflected: 0, total: particleCount });
  const [running, setRunning] = useState(true);

  // reset function (updates ref + counts)
  function resetSim() {
    particlesRef.current = new Array(particleCount).fill(0).map((_, i) => {
      const spin = Math.random() < 0.5 ? "up" : "down";
      const z = (Math.random() - 0.5) * 1.2;
      const x = left_nm + Math.random() * (Math.abs(left_nm) * 0.25);
      return { id: i, x, z, spin, decided: false, transmitted: false, speed: 0.4 + Math.random() * 0.8 };
    });
    setCounts({ transmitted: 0, reflected: 0, total: particleCount });
  }

  // ParticlesController: runs inside Canvas and updates particlesRef each frame
  function ParticlesController() {
    useFrame((state, delta) => {
      if (!running) return;
      const arr = particlesRef.current;
      const vMultiplier = 1 + 0.12 * Math.max(0, voltage - 1);
      const tScatter = 1 + (Math.max(0, temp - 25) / 200);

      for (let i = 0; i < arr.length; i++) {
        const p = arr[i];
        let x = p.x + p.speed * delta * 50 * vMultiplier / tScatter;
        let decided = p.decided;
        let transmitted = p.transmitted;

        if (!decided && p.x < barrier_start && x >= barrier_start) {
          const prob = p.spin === "up" ? Tvals.T_up : Tvals.T_down;
          const roll = Math.random();
          if (roll < prob) {
            decided = true;
            transmitted = true;
          } else {
            decided = true;
            transmitted = false;
            x = barrier_start - (x - barrier_start) - 0.5;
            p.speed = -Math.abs(p.speed) * 0.6;
          }
        }

        if (x > right_nm + 1.0) {
          const spin = Math.random() < 0.5 ? "up" : "down";
          arr[i] = {
            ...p,
            x: left_nm + Math.random() * (Math.abs(left_nm) * 0.25),
            z: (Math.random() - 0.5) * 1.2,
            spin,
            decided: false,
            transmitted: false,
            speed: 0.4 + Math.random() * 0.8,
          };
          continue;
        }

        if (x < left_nm - 1.0) {
          const spin = Math.random() < 0.5 ? "up" : "down";
          arr[i] = {
            ...p,
            x: left_nm + Math.random() * (Math.abs(left_nm) * 0.25),
            z: (Math.random() - 0.5) * 1.2,
            spin,
            decided: false,
            transmitted: false,
            speed: 0.4 + Math.random() * 0.8,
          };
          continue;
        }

        // commit updates
        arr[i].x = x;
        arr[i].decided = decided;
        arr[i].transmitted = transmitted;
      }

      // update counts less frequently? update every frame is fine for small particleCount
      const transmittedCount = arr.reduce((acc, p) => acc + ((p.transmitted || p.x > barrier_end) ? 1 : 0), 0);
      const reflectedCount = arr.reduce((acc, p) => acc + ((p.decided && !p.transmitted && p.x < barrier_start) ? 1 : 0), 0);
      setCounts({ transmitted: transmittedCount, reflected: reflectedCount, total: arr.length });
    });

    return null;
  }

  // ParticlesMesh: updates instanced meshes each frame using particlesRef
  function ParticlesMesh() {
    const upRef = useRef();
    const downRef = useRef();

    useFrame(() => {
      const arr = particlesRef.current;
      const upParticles = arr.filter((p) => p.spin === "up");
      const downParticles = arr.filter((p) => p.spin === "down");

      if (upRef.current) {
        upRef.current.count = Math.max(1, upParticles.length);
        for (let i = 0; i < upParticles.length; i++) {
          const p = upParticles[i];
          const matrix = new THREE.Matrix4();
          const pos = new THREE.Vector3(p.x * 1e-9, 0.02 + p.z * 0.02, 0);
          const scale = new THREE.Vector3(0.015, 0.015, 0.015);
          matrix.compose(pos, new THREE.Quaternion(), scale);
          upRef.current.setMatrixAt(i, matrix);
        }
        upRef.current.instanceMatrix.needsUpdate = true;
      }

      if (downRef.current) {
        downRef.current.count = Math.max(1, downParticles.length);
        for (let i = 0; i < downParticles.length; i++) {
          const p = downParticles[i];
          const matrix = new THREE.Matrix4();
          const pos = new THREE.Vector3(p.x * 1e-9, 0.02 + p.z * 0.02, 0);
          const scale = new THREE.Vector3(0.015, 0.015, 0.015);
          matrix.compose(pos, new THREE.Quaternion(), scale);
          downRef.current.setMatrixAt(i, matrix);
        }
        downRef.current.instanceMatrix.needsUpdate = true;
      }
    });

    return (
      <group>
        <instancedMesh ref={upRef} args={[null, null, Math.max(1, particleCount)]}>
          <sphereGeometry args={[1, 8, 8]} />
          <meshStandardMaterial color={"#4ea8ff"} />
        </instancedMesh>

        <instancedMesh ref={downRef} args={[null, null, Math.max(1, particleCount)]}>
          <sphereGeometry args={[1, 8, 8]} />
          <meshStandardMaterial color={"#ff6b6b"} />
        </instancedMesh>
      </group>
    );
  }

  // UI plot
  const plotData = [
    { x: ["Transmitted", "Reflected"], y: [counts.transmitted, counts.reflected], type: "bar", marker: { color: ["#4ea8ff", "#ff6b6b"] } },
  ];
  const plotLayout = { margin: { t: 10, l: 30, r: 10 }, height: 160, yaxis: { title: "count" } };

  // scene geometry metrics
  const barrierWidth_m = d_nm * 1e-9;
  const sceneWidth_m = (right_nm - left_nm) * 1e-9;
  const barrierCenterX_m = (barrier_start + barrier_end) / 2.0 * 1e-9;

  return (
    <div className="flex flex-row gap-3 h-full">
      <div className="flex-1 bg-slate-900 rounded-md p-2 h-140">
        <Canvas camera={{ position: [barrierCenterX_m + 0.0000005, 0.12, 0.0015], fov: 50 }}>
          <ambientLight intensity={0.8} />
          <directionalLight position={[0.001, 0.05, 0.05]} intensity={0.8} />

          {/* floor */}
          <mesh rotation-x={-Math.PI / 2} position={[0, -0.02, 0]}>
            <planeGeometry args={[sceneWidth_m * 1.4, 0.004]} />
            <meshStandardMaterial color={"#020617"} />
          </mesh>

          {/* barrier */}
          <mesh position={[barrierCenterX_m, 0.03, 0]} castShadow receiveShadow>
            <boxGeometry args={[barrierWidth_m, 0.06, 0.2 * barrierWidth_m + 0.02]} />
            <meshStandardMaterial color={"#888"} transparent opacity={0.9} />
          </mesh>

          {/* controller and mesh (both are children of Canvas so useFrame is valid) */}
          <ParticlesController />
          <ParticlesMesh />
        </Canvas>
      </div>

      <div className="w-72 shrink-0">
        <div className="bg-white rounded-md p-3 shadow">
          <div className="font-semibold text-sm mb-2">Transmission Summary (Tab 3)</div>

          <div className="text-xs text-slate-600 mb-2">Parameters</div>
          <ul className="text-xs">
            <li>V₀: {V0_eV.toFixed(3)} eV</li>
            <li>d: {d_nm.toFixed(3)} nm</li>
            <li>E_base: {E_eV.toFixed(4)} eV</li>
            <li>E_eff: {E_eff.toFixed(6)} eV</li>
            <li>voltage: {voltage} V</li>
            <li>temp: {temp} °C</li>
            <li>humidity: {humidity}%</li>
          </ul>

          <div className="mt-3">
            <Plot data={plotData} layout={plotLayout} config={{ displayModeBar: false }} style={{ width: "100%" }} />
          </div>

          <div className="mt-2 text-xs">
            <div>Estimated T↑: {typeof Tvals.T_up === "number" ? Tvals.T_up.toExponential(2) : String(Tvals.T_up)}</div>
            <div>Estimated T↓: {typeof Tvals.T_down === "number" ? Tvals.T_down.toExponential(2) : String(Tvals.T_down)}</div>
            <div className="mt-2">Transmitted %: {counts.total ? Math.round((counts.transmitted / counts.total) * 100) : 0}%</div>

            <div className="mt-3 flex gap-2">
              <button onClick={() => setRunning((r) => !r)} className="px-2 py-1 bg-slate-900 text-white rounded text-xs">
                {running ? "Pause" : "Resume"}
              </button>
              <button onClick={resetSim} className="px-2 py-1 bg-slate-900 text-white rounded text-xs">
                Reset
              </button>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}

import React, { useRef, useMemo, useState, useEffect } from "react";
import { useFrame, Canvas } from "@react-three/fiber";
import * as THREE from "three";
import { Html } from "@react-three/drei";
import * as PH from "./physics";
import Plot from "react-plotly.js";
function ElectronFlowSim({
  voltage = 5,
  temp = 25,
  humidity = 0, // percent 0..100
  V0_eV = 1.7,
  d_nm = 2.0,
  E_eV = 0.5,
  J_ex = 0.08,
  M_FMI = 1.0,
  particleCount = 220,
}) {
  // nm domain used by Python code (kept consistent)
  const left_nm = -2 * d_nm;
  const right_nm = 3 * d_nm;
  const barrier_start = 0;
  const barrier_end = d_nm;

  // physical helpers
  const KB_EV = 8.617333e-5; // Boltzmann constant in eV/K

  // derive effective electron energy from E_eV + voltage + temperature contributions
  const E_eff = useMemo(() => {
    // voltage adds kinetic energy (tunable coefficient)
    const vBoost = 0.02 * (Number(voltage || 0) - 1); // 0.02 eV per volt above 1V (example)
    // thermal broadening (k_B * T)
    const tBoost = KB_EV * (Number(temp || 0));
    return Number(E_eV || 0) + vBoost + tBoost;
  }, [E_eV, voltage, temp]);

  // humidity reduces transmission by scattering — clamp humidity 0..100
  const humidityFactor = useMemo(() => {
    const h = Math.max(0, Math.min(100, Number(humidity || 0)));
    // at 100% humidity reduce transmission by up to 30% (tunable)
    return 1 - (h / 100) * 0.3;
  }, [humidity]);

  // compute transmission probabilities using PH helper (applies E_eff & humidity factor)
  const Tvals = useMemo(() => {
    let tUp = 1e-12, tDown = 1e-12;
    try {
      if (typeof PH.calculateTunnelingProbabilitiesWKB === "function") {
        const res = PH.calculateTunnelingProbabilitiesWKB(V0_eV, E_eff, d_nm, J_ex, M_FMI);
        tUp = Number(res.T_up ?? res.Tup ?? res.T_up ?? 0);
        tDown = Number(res.T_down ?? res.Tdown ?? res.T_down ?? 0);
      } else {
        // fallback compute effective barrier per spin + wkbTunneling
        const DeltaE = typeof PH.spinSplitting === "function" ? PH.spinSplitting(J_ex, M_FMI) : J_ex * M_FMI;
        const V_up = V0_eV - DeltaE / 2.0;
        const V_down = V0_eV + DeltaE / 2.0;
        const d_m = d_nm * 1e-9;
        tUp = PH.wkbTunneling ? PH.wkbTunneling(V_up, d_m, E_eff) : 1e-12;
        tDown = PH.wkbTunneling ? PH.wkbTunneling(V_down, d_m, E_eff) : 1e-12;
      }
    } catch (err) {
      console.warn("Tvals calc failed:", err);
    }
    // apply humidity scattering factor and clamp 0..1
    tUp = Math.max(0, Math.min(1, tUp * humidityFactor));
    tDown = Math.max(0, Math.min(1, tDown * humidityFactor));
    return { T_up: tUp, T_down: tDown };
  }, [V0_eV, d_nm, E_eff, J_ex, M_FMI, humidityFactor]);

  // particle array state (kept in state for simplicity)
  const [particles, setParticles] = useState(() => {
    return new Array(particleCount).fill(0).map((_, i) => {
      const spin = Math.random() < 0.5 ? "up" : "down";
      const z = (Math.random() - 0.5) * 1.2;
      const x = left_nm + Math.random() * (Math.abs(left_nm) * 0.25);
      return {
        id: i,
        x,
        z,
        spin,
        decided: false,
        transmitted: false,
        speed: 0.4 + Math.random() * 0.8,
      };
    });
  });

  // counts for UI
  const [counts, setCounts] = useState({ transmitted: 0, reflected: 0, total: particleCount });
  const [running, setRunning] = useState(true);

  // reset helper
  function resetSim() {
    setParticles(
      new Array(particleCount).fill(0).map((_, i) => {
        const spin = Math.random() < 0.5 ? "up" : "down";
        const z = (Math.random() - 0.5) * 1.2;
        const x = left_nm + Math.random() * (Math.abs(left_nm) * 0.25);
        return { id: i, x, z, spin, decided: false, transmitted: false, speed: 0.4 + Math.random() * 0.8 };
      })
    );
    setCounts({ transmitted: 0, reflected: 0, total: particleCount });
  }

  // animate particles and update setParticles (keeps logic similar to your prior version)
  useFrame((state, delta) => {
    if (!running) return;
    setParticles((prev) => {
      const newArr = prev.map((p) => {
        // speed modulation by voltage & temp (voltage increases speed, high temp increases spread)
        const vMultiplier = 1 + 0.12 * Math.max(0, voltage - 1); // tuneable
        const tScatter = 1 + (Math.max(0, temp - 25) / 200); // >25ºC slightly increases dispersion
        let x = p.x + p.speed * delta * 50 * vMultiplier / tScatter;
        let decided = p.decided;
        let transmitted = p.transmitted;

        // when reaching barrier start, make probabilistic decision
        if (!decided && p.x < barrier_start && x >= barrier_start) {
          const prob = p.spin === "up" ? Tvals.T_up : Tvals.T_down;
          const roll = Math.random();
          if (roll < prob) {
            decided = true;
            transmitted = true;
          } else {
            decided = true;
            transmitted = false;
            // reflect: bounce slightly and reverse speed
            x = barrier_start - (x - barrier_start) - 0.5;
            return { ...p, x, z: p.z, spin: p.spin, decided, transmitted, speed: -Math.abs(p.speed) * 0.6 };
          }
        }

        // respawn if far right
        if (x > right_nm + 1.0) {
          const spin = Math.random() < 0.5 ? "up" : "down";
          return {
            ...p,
            x: left_nm + Math.random() * (Math.abs(left_nm) * 0.25),
            z: (Math.random() - 0.5) * 1.2,
            spin,
            decided: false,
            transmitted: false,
            speed: 0.4 + Math.random() * 0.8,
          };
        }

        // respawn if far left
        if (x < left_nm - 1.0) {
          const spin = Math.random() < 0.5 ? "up" : "down";
          return {
            ...p,
            x: left_nm + Math.random() * (Math.abs(left_nm) * 0.25),
            z: (Math.random() - 0.5) * 1.2,
            spin,
            decided: false,
            transmitted: false,
            speed: 0.4 + Math.random() * 0.8,
          };
        }

        return { ...p, x, z: p.z, spin: p.spin, decided, transmitted, speed: p.speed };
      });

      // recompute counts (cheap)
      const transmittedCount = newArr.reduce((acc, p) => acc + ((p.transmitted || p.x > barrier_end) ? 1 : 0), 0);
      const reflectedCount = newArr.reduce((acc, p) => acc + ((p.decided && !p.transmitted && p.x < barrier_start) ? 1 : 0), 0);
      setCounts({ transmitted: transmittedCount, reflected: reflectedCount, total: newArr.length });

      return newArr;
    });
  });

  // ParticlesMesh: two instancedMesh groups (up/down)
  function ParticlesMesh({ particles }) {
    const upRef = useRef();
    const downRef = useRef();

    useFrame(() => {
      // fill up mesh with up-spin particles
      if (upRef.current) {
        const upParticles = particles.filter((p) => p.spin === "up");
        upRef.current.count = upParticles.length;
        for (let i = 0; i < upParticles.length; i++) {
          const p = upParticles[i];
          const matrix = new THREE.Matrix4();
          const pos = new THREE.Vector3(p.x * 1e-9, 0.02 + p.z * 0.02, 0);
          const scale = new THREE.Vector3(0.015, 0.015, 0.015);
          matrix.compose(pos, new THREE.Quaternion(), scale);
          upRef.current.setMatrixAt(i, matrix);
        }
        upRef.current.instanceMatrix.needsUpdate = true;
      }

      // fill down mesh
      if (downRef.current) {
        const downParticles = particles.filter((p) => p.spin === "down");
        downRef.current.count = downParticles.length;
        for (let i = 0; i < downParticles.length; i++) {
          const p = downParticles[i];
          const matrix = new THREE.Matrix4();
          const pos = new THREE.Vector3(p.x * 1e-9, 0.02 + p.z * 0.02, 0);
          const scale = new THREE.Vector3(0.015, 0.015, 0.015);
          matrix.compose(pos, new THREE.Quaternion(), scale);
          downRef.current.setMatrixAt(i, matrix);
        }
        downRef.current.instanceMatrix.needsUpdate = true;
      }
    });

    // two separate instance meshes so color is simple
    return (
      <group>
        <instancedMesh ref={upRef} args={[null, null, Math.max(1, particles.filter(p => p.spin === "up").length)]}>
          <sphereGeometry args={[1, 8, 8]} />
          <meshStandardMaterial color={"#4ea8ff"} />
        </instancedMesh>

        <instancedMesh ref={downRef} args={[null, null, Math.max(1, particles.filter(p => p.spin === "down").length)]}>
          <sphereGeometry args={[1, 8, 8]} />
          <meshStandardMaterial color={"#ff6b6b"} />
        </instancedMesh>
      </group>
    );
  }

  // Plotly data
  const plotData = [
    {
      x: ["Transmitted", "Reflected"],
      y: [counts.transmitted, counts.reflected],
      type: "bar",
      marker: { color: ["#4ea8ff", "#ff6b6b"] },
    },
  ];
  const plotLayout = { margin: { t: 10, l: 30, r: 10 }, height: 160, yaxis: { title: "count" } };

  // convert barrier dims to meters for the 3D box
  const barrierWidth_m = d_nm * 1e-9;
  const sceneWidth_m = (right_nm - left_nm) * 1e-9;
  const barrierCenterX_m = (barrier_start + barrier_end) / 2.0 * 1e-9;

  return (
    <div className="flex flex-row gap-3 h-full">
      {/* Left: Canvas */}
      <div className="flex-1 bg-slate-900 rounded-md p-2 h-140">
        <Canvas camera={{ position: [barrierCenterX_m + 0.0000005, 0.12, 0.0015], fov: 50 }}>
          <ambientLight intensity={0.8} />
          <directionalLight position={[0.001, 0.05, 0.05]} intensity={0.8} />
          {/* floor */}
          <mesh rotation-x={-Math.PI / 2} position={[0, -0.02, 0]}>
            <planeGeometry args={[sceneWidth_m * 1.4, 0.004]} />
            <meshStandardMaterial color={"#020617"} />
          </mesh>

          {/* barrier box */}
          <mesh position={[barrierCenterX_m, 0.03, 0]} castShadow receiveShadow>
            <boxGeometry args={[barrierWidth_m, 0.06, 0.2 * barrierWidth_m + 0.02]} />
            <meshStandardMaterial color={"#888"} transparent opacity={0.9} />
          </mesh>

          {/* Particles (instanced) */}
          <ParticlesMesh particles={particles} />
        </Canvas>
      </div>

      {/* Right: stats */}
      <div className="w-72 shrink-0">
        <div className="bg-white rounded-md p-3 shadow">
          <div className="font-semibold text-sm mb-2">Transmission Summary (Tab 3)</div>

          <div className="text-xs text-slate-600 mb-2">Parameters</div>
          <ul className="text-xs">
            <li>V₀: {V0_eV.toFixed(3)} eV</li>
            <li>d: {d_nm.toFixed(3)} nm</li>
            <li>E_base: {E_eV.toFixed(4)} eV</li>
            <li>E_eff: {E_eff.toFixed(6)} eV</li>
            <li>voltage: {voltage} V</li>
            <li>temp: {temp} °C</li>
            <li>humidity: {humidity}%</li>
          </ul>

          <div className="mt-3">
            <Plot data={plotData} layout={plotLayout} config={{ displayModeBar: false }} style={{ width: "100%" }} />
          </div>

          <div className="mt-2 text-xs">
            <div>
              Estimated T↑: {typeof Tvals.T_up === "number" ? Tvals.T_up.toExponential(2) : String(Tvals.T_up)}
            </div>
            <div>
              Estimated T↓: {typeof Tvals.T_down === "number" ? Tvals.T_down.toExponential(2) : String(Tvals.T_down)}
            </div>
            <div className="mt-2">Transmitted %: {counts.total ? Math.round((counts.transmitted / counts.total) * 100) : 0}%</div>

            <div className="mt-3 flex gap-2">
              <button onClick={() => setRunning((r) => !r)} className="px-2 py-1 bg-slate-900 text-white rounded text-xs">
                {running ? "Pause" : "Resume"}
              </button>
              <button onClick={resetSim} className="px-2 py-1 bg-slate-900 text-white rounded text-xs">
                Reset
              </button>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
}
export default ElectronFlowSim;

  function spawnGhost({ x_nm, z, prob, colorHex = 0xffffff, opacityScale = 1, strong = false }) {
    const ghosts = ghostsRef.current;
    let idx = ghosts.findIndex((g) => g === null || g.life <= 0);
    if (idx === -1) {
      idx = 0; // recycle oldest
    }
    const base = Math.max(0.02, Math.min(0.95, prob * opacityScale));
    ghosts[idx] = {
      id: `g-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
      x: x_nm,
      z,
      life: strong ? 1.2 : 0.9,
      opacity: strong ? Math.min(1, base * 1.8) : base * 0.18,
      size: strong ? 0.06 + prob * 0.12 : 0.02 + prob * 0.06,
      colorHex,
    };
  }

    fgChargesRef.current.push({ id: `fg-${Date.now()}-${Math.random().toString(36).slice(2, 5)}`, x, z, age: 0 });
  }
